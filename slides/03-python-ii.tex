sum = 0.0
sum2 = 0.0
for v in values:
    sum = sum + v
    sum2 = sum2 + v * v

mu2 = sum2/len(values)
print 'Average:', mu

gcd(a,b) = \begin{cases}

            gcd(a-b, b) & \text{if $a > b$} \\
            gcd(a, b-a) & \text{o.w.}
            \end{cases}
\]
\end{block}
