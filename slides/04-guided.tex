\begin{frame}[fragile]
\frametitle{How Many Above 18?}

\begin{python}
ages = [10, 54, 13, 90, 13, 18]
\end{python}

How many are above 18?

\pause

\begin{python}
above18 = 0
for v in ages:
    if v > 18:
        above18 += 1
print above18
\end{python}

\end{frame}


\begin{frame}[fragile]
\frametitle{What does this print?}

\begin{python}
ages = [10, 54, 23, 90, 23, 18]

while ages[vi] <= 18:
    vi += 1
print vi
\end{python}

Prints

\begin{enumerate}
\item 0
\item 1
\item 2
\item 54
\end{enumerate}

\end{frame}
